37 GLubyte* TexturedPolygons::LoadRawImageFile(char* filename, int width, int height)
38 {
39  FILE* file;
40  unsigned char* image;
41  // create memory space w x h x 3 (3 stores RGB values)
42  image = new unsigned char[width * height * 3];
43  file = fopen(filename, "rb");
44  // exit program if image not found and inform user
45  if (file == nullptr)
46  {
47  std::cout << "ERROR loading image file: " << filename << "...\n";
48  exit(0);
49  }
50  fread(image, width * height * 3, 1, file);
51  fclose(file);
52  return image;

83 void TexturedPolygons::CreateDisplayList(const int &XYZ, const int &listNo,
84  const GLdouble &xImgSize, const GLdouble &zImgSize,
85  const GLdouble &xStart, const GLdouble &yStart, const GLdouble &zStart,
86  const GLdouble &xTimes, const GLdouble &zTimes)
87 {
88  glNewList(listNo, GL_COMPILE);
89  glBegin(GL_QUADS);
90  switch (XYZ)
91  {
92  case 0:
93  // create list where image is on X and Y axis
94  CreateXtoYTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes, false);
95  break;
96  case 1:
97  // create list where image is on X and Z axis
98  CreateXtoZTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes);
99  break;
100  case 2:
101  // create list where image is on Y and Z axis
102  CreateYtoZTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes, false);
103  break;
104  case 3:
105  // create list where (flipped) image is on X and Z axis
106  CreateYtoZTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes, true);
107  break;
108  case 4:
109  // create list where (flipped) image is on X and Y axis
110  CreateXtoYTextureList(xImgSize, zImgSize, xStart, yStart, zStart, xTimes, zTimes, true);
111  break;
112  }
113  glEnd();
114  glEndList();
115 }

139 void TexturedPolygons::CreateXtoYTextureList(const GLdouble &xImgSize, const GLdouble &yImgSize,
