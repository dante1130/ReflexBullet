121 void Camera::WSKeyboardMovement()
122 {
123  if (m_deltaMoveFB == 0) { return; }
124 
125  float movementSpeed = m_moveSpeed;
126  if (m_deltaMoveLR != 0) { movementSpeed *= 0.5; } //So you can't run at twice the speed when running diagonally
127  if (crouch) { movementSpeed *= 0.5; }
128 
129  float xMove = m_look.x * movementSpeed;
130  float zMove = m_look.z* movementSpeed;
131 
132  float norm = movementSpeed / sqrt(pow(xMove, 2) + pow(zMove, 2)); //So it runs normally when looking up or down
133  xMove *= norm * m_deltaMoveFB; //m_deltaMoveFB is used for the direction
134  zMove *= norm * m_deltaMoveFB;
135 
136  //Checks if player can move in direction based on AABB
137  if (!(m_colDetect.Collide(glm::dvec3(m_pos.x + xMove, m_pos.y + m_lookK.y, m_pos.z))))
138  {
139  m_pos.x += xMove;
140  }
141 
142  if (!(m_colDetect.Collide(glm::dvec3(m_pos.x, m_pos.y + m_lookK.y, m_pos.z + zMove))))
143  {
144  m_pos.z += zMove;
145  }
146 
147  //Makes sure that the camera object is on the right y height
148  SetPlains(xMove, zMove);
149 
150 }

251 void Camera::MouseMove(int x, int y)
252 {
253  if (m_firstMouse)
254  {
255  m_prev.x = 103.55;
256  m_prev.y = y;
257  m_firstMouse = false;
258  }
259 
260  // calculate delta by offsetting it with the previous x and y values with new values
261  m_deltaAngleLR = x - m_prev.x;
262  m_deltaAngleUD = m_prev.y - y;
263 
264  // updates the previous x and y values
265  m_prev.x = x;
266  m_prev.y = y;
267 
268  // updates the rotation angle
269  m_rotateAngleLR += m_deltaAngleLR;
270  m_rotateAngleUD += m_deltaAngleUD;
271 
272  // clamps between rotation angles of 89 and -89
273  if (m_rotateAngleUD > 89.0f)
274  m_rotateAngleUD = 89.0f;
275  else if (m_rotateAngleUD < -89.0f)
276  m_rotateAngleUD = -89.0f;
277 
278  m_look.x = sin(degreesToRadians(m_rotateAngleLR)) * cos(degreesToRadians(m_rotateAngleUD));
279  m_look.y = sin(degreesToRadians(m_rotateAngleUD));
280  m_look.z = -cos(degreesToRadians(m_rotateAngleLR)) * cos(degreesToRadians(m_rotateAngleUD));
281 
282  int winW = glutGet(GLUT_WINDOW_WIDTH);
283  int winH = glutGet(GLUT_WINDOW_HEIGHT);
284 
285  m_prev.x = winW / 2;
286  m_prev.y = winH / 2;
287  glutWarpPointer(m_prev.x, m_prev.y); //centers the cursor
288 
289  callGLLookAt();
290 }

303 void Camera::SetPlains(const int & moveX, const int & moveZ)
304 {
305  // store number of plains (stops from looping through linked list each time)
306  if (m_No_Plains == 0) m_No_Plains = m_plain.Size();
307 
308 
309  glm::vec3 plainStart, plainEnd;
310  float x1, x2, z1, z2, dif, dist, ratio, area, areaSum;
311 
312  for (int i = 0; i < m_No_Plains; i++)
313  {
314  plainStart = m_plain.GetStart(i);
315  plainEnd = m_plain.GetEnd(i);
316 
317  // if camera is positioned on a plain
318  if ((m_pos.z <= plainEnd.z) && (m_pos.z >= plainStart.z)
319  && (m_pos.x <= plainEnd.x) && (m_pos.x >= plainStart.x))
320  {
321  // if flat plain
322  if (m_plain.GetType(i) == 0)
323  {
324  m_pos.y = plainEnd.y;
325 
326  if ((m_plainNo != i) && m_plainHeight != plainStart.y)
327  {
328  m_audio.PlaySound("stairstep");
329  }
330 
331  m_plainNo = i;
332  m_plainHeight = plainStart.y;
333  }
334  // if plain slopes in z direction
335  if (m_plain.GetType(i) == 2)
336  {
337  z1 = plainStart.z;
338  z2 = plainEnd.z;
339  dif = z1 - z2;
340 
341  dist = m_pos.z - z1;
342  ratio = (plainStart.y - plainEnd.y) / dif;
343 
344  m_pos.y = plainStart.y + ratio * dist;
345  }
346  // if plain slopes in x direction
347  if (m_plain.GetType(i) == 1)
348  {
349  x1 = plainStart.x;
350  x2 = plainEnd.x;
351  dif = x1 - x2;
352 
353  dist = m_pos.x - x1;
354  ratio = (plainStart.y - plainEnd.y) / dif;
355 
356  m_pos.y = plainStart.y + ratio * dist;
357  }
358 
359  //if plain is diagonal flat plain
360 
361  }
362  else if (m_plain.GetType(i) == DIAGONAL_FLAT_PLAIN)
363  {
364  plainStart = m_plain.GetQ2(i) - m_plain.GetStart(i);
365  plainEnd = m_plain.GetEnd(i) - m_plain.GetQ2(i);
366  area = abs(plainStart.z * plainEnd.x - plainStart.x * plainEnd.z);
367 
368  plainStart = m_plain.GetQ4(i) - m_plain.GetEnd(i);
369  plainEnd = m_plain.GetQ4(i) - m_plain.GetStart(i);
370  area += abs(plainStart.z * plainEnd.x - plainStart.x * plainEnd.z);
371 
372  areaSum = AreaPlainTriangle(1, i);
373  areaSum += AreaPlainTriangle(2, i);
374  areaSum += AreaPlainTriangle(3, i);
375  areaSum += AreaPlainTriangle(4, i);
376 
377  if (areaSum - area >= -0.01 && areaSum - area <= 0.01)
378  {
379  m_pos.y = m_plain.GetStart(i).y;
380  if ((m_plainNo != i) && m_plainHeight != m_plain.GetStart(i).y)
381  {
382  m_audio.PlaySound("stairstep");
383  }
384  m_plainNo = i;
385  m_plainHeight = m_plain.GetStart(i).y;
386  }
387  }
