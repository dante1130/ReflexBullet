246  GLfloat playerAngle = 180 - acos((radius * radius + bc * bc - ac * ac) / (2 * radius * bc)) * 180.0 / M_PI; //angle between abc
247  if (player_Pos.z - m_position.z > 0)
248  playerAngle = (180 - playerAngle) + 180;
