107 void Camera::WSKeyboardMovement()
108 {
109  if (m_deltaMoveFB == 0) { return; }
110 
111  float movementSpeed = m_moveSpeed;
112  if (m_deltaMoveLR != 0) { movementSpeed *= 0.5; } //So you can't run at twice the speed when running diagonally
113  if (crouch) { movementSpeed *= 0.5; }
114 
115  float xMove = m_look.x * movementSpeed;
116  float zMove = m_look.z* movementSpeed;
117 
118  float norm = movementSpeed / sqrt(pow(xMove, 2) + pow(zMove, 2)); //So it runs normally when looking up or down
119  xMove *= norm * m_deltaMoveFB; //m_deltaMoveFB is used for the direction
120  zMove *= norm * m_deltaMoveFB;
121 
122  //Checks if player can move in direction based on AABB
123  if (!(m_colDetect.Collide(glm::dvec3(m_pos.x + xMove, m_pos.y + m_lookK.y, m_pos.z))))
124  {
125  m_pos.x += xMove;
126  }
127 
128  if (!(m_colDetect.Collide(glm::dvec3(m_pos.x, m_pos.y + m_lookK.y, m_pos.z + zMove))))
129  {
130  m_pos.z += zMove;
131  }
132 
133  //Makes sure that the camera object is on the right y height
134  SetPlains(xMove, zMove);
135 
136 }

226 void Camera::MouseMove(int x, int y)
227 {
228  if (m_firstMouse)
229  {
230  m_prev.x = 103.55;
231  m_prev.y = y;
232  m_firstMouse = false;
233  }
234 
235  // calculate delta by offsetting it with the previous x and y values with new values
236  m_deltaAngleLR = x - m_prev.x;
237  m_deltaAngleUD = m_prev.y - y;
238 
239  // updates the previous x and y values
240  m_prev.x = x;
241  m_prev.y = y;
242 
243  // updates the rotation angle
244  m_rotateAngleLR += m_deltaAngleLR * m_rotateSpeed;
245  m_rotateAngleUD += m_deltaAngleUD * m_rotateSpeed;
246 
247  // clamps between rotation angles of 89 and -89
248  if (m_rotateAngleUD > 89.0f)
249  m_rotateAngleUD = 89.0f;
250  else if (m_rotateAngleUD < -89.0f)
251  m_rotateAngleUD = -89.0f;
252 
253  m_look.x = sin(degreesToRadians(m_rotateAngleLR)) * cos(degreesToRadians(m_rotateAngleUD));
254  m_look.y = sin(degreesToRadians(m_rotateAngleUD));
255  m_look.z = -cos(degreesToRadians(m_rotateAngleLR)) * cos(degreesToRadians(m_rotateAngleUD));
256 
257  int winW = glutGet(GLUT_WINDOW_WIDTH);
258  int winH = glutGet(GLUT_WINDOW_HEIGHT);
259 
260  m_prev.x = winW / 2;
261  m_prev.y = winH / 2;
262  glutWarpPointer(m_prev.x, m_prev.y); //centers the cursor
263 
264  callGLLookAt();
265 }

278 void Camera::SetPlains(const int & moveX, const int & moveZ)
279 {
280  // store number of plains (stops from looping through linked list each time)
281  if (m_No_Plains == 0) m_No_Plains = m_plain.Size();
282 
283 
284  glm::vec3 plainStart, plainEnd;
285  float x1, x2, z1, z2, dif, dist, ratio, area, areaSum;
286 
287  for (int i = 0; i < m_No_Plains; i++)
288  {
289  plainStart = m_plain.GetStart(i);
290  plainEnd = m_plain.GetEnd(i);
291 
292  // if camera is positioned on a plain
293  if ((m_pos.z <= plainEnd.z) && (m_pos.z >= plainStart.z)
294  && (m_pos.x <= plainEnd.x) && (m_pos.x >= plainStart.x))
295  {
296  // if flat plain
297  if (m_plain.GetType(i) == 0)
298  {
299  m_pos.y = plainEnd.y;
300 
301  if ((m_plainNo != i) && m_plainHeight != plainStart.y)
302  {
303  m_audio.PlaySound("stairstep");
304  }
305 
306  m_plainNo = i;
307  m_plainHeight = plainStart.y;
308  }
309  // if plain slopes in z direction
310  if (m_plain.GetType(i) == 2)
311  {
312  z1 = plainStart.z;
313  z2 = plainEnd.z;
314  dif = z1 - z2;
315 
316  dist = m_pos.z - z1;
317  ratio = (plainStart.y - plainEnd.y) / dif;
318 
319  m_pos.y = plainStart.y + ratio * dist;
320  }
321  // if plain slopes in x direction
322  if (m_plain.GetType(i) == 1)
323  {
324  x1 = plainStart.x;
325  x2 = plainEnd.x;
326  dif = x1 - x2;
327 
328  dist = m_pos.x - x1;
329  ratio = (plainStart.y - plainEnd.y) / dif;
330 
331  m_pos.y = plainStart.y + ratio * dist;
332  }
333 
334  //if plain is diagonal flat plain
335 
336  }
337  else if (m_plain.GetType(i) == DIAGONAL_FLAT_PLAIN)
338  {
339  plainStart = m_plain.GetQ2(i) - m_plain.GetStart(i);
340  plainEnd = m_plain.GetEnd(i) - m_plain.GetQ2(i);
341  area = abs(plainStart.z * plainEnd.x - plainStart.x * plainEnd.z);
342 
343  plainStart = m_plain.GetQ4(i) - m_plain.GetEnd(i);
344  plainEnd = m_plain.GetQ4(i) - m_plain.GetStart(i);
345  area += abs(plainStart.z * plainEnd.x - plainStart.x * plainEnd.z);
346 
347  areaSum = AreaPlainTriangle(1, i);
348  areaSum += AreaPlainTriangle(2, i);
349  areaSum += AreaPlainTriangle(3, i);
350  areaSum += AreaPlainTriangle(4, i);
351 
352  if (areaSum - area >= -0.01 && areaSum - area <= 0.01)
353  {
354  m_pos.y = m_plain.GetStart(i).y;
355  if ((m_plainNo != i) && m_plainHeight != m_plain.GetStart(i).y)
356  {
357  m_audio.PlaySound("stairstep");
358  }
359  m_plainNo = i;
360  m_plainHeight = m_plain.GetStart(i).y;
361  }
362  }
363  }
364 }
